Simple Configuration, Powerful Results

# load-test.yml - Simple load test configuration
name: "E-commerce Load Test"

global:
  base_url: "https://api.example.com"

load:
  pattern: "basic"
  virtual_users: 50
  duration: "5m"
  ramp_up: "30s"

scenarios:
  - name: "user_journey"
    steps:
      - name: "login"
        type: "rest"
        method: "POST"
        path: "/auth/login"
        json:
          email: "{{faker.internet.email}}"
          password: "{{faker.internet.password}}"
        checks:
          - type: "status"
            value: 200
// Advanced TypeScript scenario with fluent DSL
import { test, load, testData } from 'perfornium';

const config = test('E-commerce Load Test')
  .baseUrl('https://api.example.com')
  .withBrowser('chromium', { headless: true })
  .scenario('User Journey')
    .post('/auth/login', {
      email: testData.email(),
      password: testData.password()
    })
    .check('status', 200)
    .extract('token', '$.access_token')
    .goto('/products')
    .click('[data-testid="add-to-cart"]')
    .expectVisible('#cart-count')
    .done()
  .withLoad(load().virtualUsers(50).duration('5m'))
  .withReport('reports/load-test.html')
  .build();
# Initialize a new test project
perfornium init --template api

# Record user interactions to generate test scenarios
perfornium record http://localhost:3000 -o tests/web/login.yml

# Import OpenAPI spec to generate tests
perfornium import openapi api-spec.yaml -o tests/api

# Run a performance test with HTML report
perfornium run tests/load-test.yml --report

# Start a worker node for distributed testing
perfornium worker -p 8080

# Run distributed test across multiple workers
perfornium distributed tests/load-test.yml -w 192.168.1.10:8080,192.168.1.11:8080
